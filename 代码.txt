一、数据生成代码（本地 Python 运行）
import pandas as pd
import numpy as np

np.random.seed(42)

n_normal = 150
n_fault = 150
n_features = 30

normal_data = np.random.normal(loc=0, scale=1, size=(n_normal, n_features))
normal_label = np.zeros((n_normal, 1))

fault_data = np.random.normal(loc=1.8, scale=1.5, size=(n_fault, n_features))
fault_label = np.ones((n_fault, 1))

data = np.vstack((np.hstack((normal_label, normal_data)),
                  np.hstack((fault_label, fault_data))))
np.random.shuffle(data)

columns = ['label'] + [f'feature_{i}' for i in range(1, n_features + 1)]
df = pd.DataFrame(data, columns=columns)
df.to_csv('motor_data.csv', index=False)

print(f"数据集已生成：motor_data.csv，共 {len(df)} 行，{len(df.columns)} 列")

四、数据清洗代码（PyODPS 节点）
from odps import ODPS
import numpy as np
import pandas as pd

# 获取ODPS对象
o = get_odps()

# 读取源表
table = o.get_table('motor_data_ods')
df = table.to_pandas(limit=None)
print(f"原始数据行数: {len(df)}")

# 3σ截断清洗
feature_cols = [f'feature_{i}' for i in range(1, 31)]
for col in feature_cols:
    mean = df[col].mean()
    std = df[col].std()
    lower = mean - 3 * std
    upper = mean + 3 * std
    df[col] = df[col].clip(lower, upper)

# 定义表结构
table_schema = """
label bigint, feature_1 double, feature_2 double, feature_3 double,
feature_4 double, feature_5 double, feature_6 double,
feature_7 double, feature_8 double, feature_9 double,
feature_10 double, feature_11 double, feature_12 double,
feature_13 double, feature_14 double, feature_15 double,
feature_16 double, feature_17 double, feature_18 double,
feature_19 double, feature_20 double, feature_21 double,
feature_22 double, feature_23 double, feature_24 double,
feature_25 double, feature_26 double, feature_27 double,
feature_28 double, feature_29 double, feature_30 double
"""

# 创建表
o.create_table('motor_data_cleaned', table_schema, if_not_exists=True)

# 写入数据
o.write_table('motor_data_cleaned', df.values.tolist(), overwrite=True)
print(f"清洗后数据行数: {len(df)}")
